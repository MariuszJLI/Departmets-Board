const proxyBase = "https://departments-board.glitch.me"; const apiUrlMainData = `${proxyBase}/?action=getMain`; const apiUrlModalData = `${proxyBase}/?action=getModal`; const apiUrlCountData = `${proxyBase}/?action=getCounts`; const apiUrlUpdateCell = proxyBase; let selectedCell = null; let addRowDepartment = null; let newEmployeeName = null; let availableNames = []; let availableTaskNames = {}; let currentModalTasks = []; async function fetchAndDisplay() { const [mainRes, modalRes, countRes] = await Promise.all([ fetch(apiUrlMainData), fetch(apiUrlModalData), fetch(apiUrlCountData) ]); const main = await mainRes.json(); const modal = await modalRes.json(); const counts = await countRes.json(); availableNames = modal.employees || []; availableTaskNames = { distribution: modal.distributionTasks || [], goodsin: modal.goodsInTasks || [], pallets: modal.palletsTasks || [] }; document.getElementById("datetime").textContent = "Last updated: " + new Date().toLocaleString(); document.getElementById("total-count").textContent = `Total Employees: ${counts.total}`; ["distribution", "goodsin", "pallets"].forEach(dept => { const table = document.getElementById(`${dept}-table`); const tbody = table.querySelector("tbody"); tbody.innerHTML = ""; document.getElementById(`${dept}-leader`).textContent = `Leader: ${main.leaders?.[dept] || "N/A"}`; document.getElementById(`${dept}-count`).textContent = `Employees: ${counts[dept] || 0}`; (main[dept] || []).forEach(([name, task], i) => { const row = document.createElement("tr"); const nameCell = document.createElement("td"); const taskCell = document.createElement("td"); const delCell = document.createElement("td"); nameCell.textContent = name; taskCell.textContent = task || "-"; delCell.textContent = "â€“"; delCell.style.color = "red"; delCell.style.textAlign = "center"; delCell.style.cursor = "pointer"; nameCell.onclick = () => openEditModal("name", dept, i + 12, nameCell); taskCell.onclick = () => openEditModal("task", dept, i + 12, taskCell); delCell.onclick = () => deleteRow(dept, i + 12); row.append(nameCell, taskCell, delCell); tbody.appendChild(row); }); const addBtn = document.getElementById(`${dept}-add`); if (!addBtn) { const button = document.createElement("button"); button.id = `${dept}-add`; button.textContent = "+ Add Row"; button.style.marginTop = "10px"; button.onclick = () => openAddNameModal(dept); table.closest(".department").appendChild(button); } }); populateModal("nameList", availableNames, (name) => { newEmployeeName = name; closeModal("nameModal"); openAddTaskModal(); }); document.getElementById("nameSearch").oninput = () => filterList("nameSearch", "nameList"); document.getElementById("taskSearch").oninput = () => filterList("taskSearch", "taskList"); } function populateModal(listId, values, onClickFn) { const list = document.getElementById(listId); list.innerHTML = ""; values.forEach(item => { const li = document.createElement("li"); li.textContent = item; li.onclick = () => onClickFn(item); list.appendChild(li); }); } function openEditModal(type, dept, row, cell) { selectedCell = cell; selectedCell.dataset.colType = type; selectedCell.dataset.department = dept; selectedCell.dataset.rowIndex = row; if (type === "name") { document.getElementById("nameModal").style.display = "block"; } else { populateModal("taskList", availableTaskNames[dept], (task) => { selectedCell.textContent = task; updateSheetValue("task", task); closeModal("taskModal"); }); document.getElementById("taskModal").style.display = "block"; } } function openAddNameModal(dept) { addRowDepartment = dept; newEmployeeName = null; document.getElementById("nameModal").style.display = "block"; } function openAddTaskModal() { currentModalTasks = availableTaskNames[addRowDepartment] || []; populateModal("taskList", currentModalTasks, (task) => { addNewRowToSheet(addRowDepartment, newEmployeeName, task); closeModal("taskModal"); }); document.getElementById("taskModal").style.display = "block"; } function addNewRowToSheet(department, name, task) { fetch(apiUrlUpdateCell, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ action: "addRow", department, name, task }) }).then(() => fetchAndDisplay()); } function updateSheetValue(colType, value) { const { department, rowIndex } = selectedCell.dataset; fetch(apiUrlUpdateCell, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ department, rowIndex, colType, value }) }).then(() => fetchAndDisplay()); } function deleteRow(department, rowIndex) { fetch(apiUrlUpdateCell, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ action: "deleteRow", department, rowIndex }) }).then(() => fetchAndDisplay()); } function filterList(searchId, listId) { const filter = document.getElementById(searchId).value.toLowerCase(); document.querySelectorAll(`#${listId} li`).forEach(li => { li.style.display = li.textContent.toLowerCase().includes(filter) ? "" : "none"; }); } function closeModal(id) { document.getElementById(id).style.display = "none"; } document.getElementById("nameClose").onclick = () => closeModal("nameModal"); document.getElementById("taskClose").onclick = () => closeModal("taskModal"); fetchAndDisplay(); setInterval(fetchAndDisplay, 60000);
